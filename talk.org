** Fibonacci

#+BEGIN_SRC emacs-lisp
   (defun fib (n)
     (cond ((= n 0) 0)
           ((= n 1) 1)
           (t (+ (fib (- n 1))
                 (fib (- n 2))))))
#+END_SRC

** Make it interactive

#+BEGIN_SRC emacs-lisp
  (defun fib-interactive
      (number)
    (interactive "nNumber:\n")
    (message (format "Result is %s" (fib number))))
#+END_SRC
